JS:

'use strict'
document.addEventListener('DOMContentLoaded', function () {

  // полифил метода forEach() для NodeList
  if (window.NodeList && !NodeList.prototype.forEach) {
    NodeList.prototype.forEach = function (callback, thisArg) {
      thisArg = thisArg || window;
      for (var i = 0; i < this.length; i++) {
        callback.call(thisArg, this[i], i, this);
      }
    }
  }

  /* находим все обёртки каждого селекта  */
  const dropDown = document.querySelectorAll('.dropdown');

  /* перебираем каждую обёртку и задаём аргумент */
  dropDown.forEach(function (dropDownWrapper) {

    /* далее в каждой обёртке находим кнопку, список с элементами внутри, поле ввода и также присваиваем им переменные */
    const dropDownBtn = dropDownWrapper.querySelector('.dropdown__button');
    const dropDownList = dropDownWrapper.querySelector('.dropdown__list');
    const dropDownListItems = dropDownList.querySelectorAll('.dropdown__list-item');
    const dropDownInput = dropDownWrapper.querySelector('.dropdown__input-hidden');

    /* при клике на кнопку открывается/закрывается список */
    dropDownBtn.addEventListener('click', function () {
      dropDownList.classList.toggle('dropdown__list--visible');

      /* присваивается активный класс кнопке (самодельный фокус) */
      this.classList.add('dropdown__button--active');
    });

    /* при клике на отдельный элемент списка закрывается список и выбирается значение элемента для кнопки */
    dropDownListItems.forEach(function (listItem) {
      listItem.addEventListener('click', function (e) {

        e.stopPropagation(); /* остановка передачи события наверх,
        т.е., клик не считается сторонним и класс dropdown__button--active не удаляется */

        dropDownBtn.innerText = this.innerText;

        dropDownList.classList.remove('dropdown__list--visible');

        /* кнопка остаётся в фокусе даже при нажатии на элемент списка */
        dropDownBtn.focus();

        /* присваивается значение элемента списка для инпута */
        dropDownInput.value = this.dataset.value;
      });
    });

    /* проверяется клик вне кнопки. если true, то закрывается */
    document.addEventListener("click", function (e) {
      if (e.target !== dropDownBtn) {
        dropDownBtn.classList.remove('dropdown__button--active');
        dropDownList.classList.remove('dropdown__list--visible');
      }
    });

    /* при нажатии на кнопку Tab или Escape активный класс кнопки (самодельный фокус) удаляется */
    document.addEventListener('keydown', function (e) {
      if (e.key === "Tab" || e.key === "Escape") {
        dropDownBtn.classList.remove('dropdown__button--active');
        dropDownList.classList.remove('dropdown__list--visible');
      }
    });

  });

});
